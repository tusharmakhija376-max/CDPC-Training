1. Palindrome Number

This program checks whether a given number is a palindrome.

class Solution(object):
    def isPalindrome(self, x):
        original = x
        reverse = 0

        while x > 0:
            digit = x % 10
            reverse = reverse * 10 + digit
            x = x // 10

        return reverse == original

Example
Input: 121  
Output: True


2. Richest Customer Wealth

This program finds the maximum wealth among all customers.

class Solution(object):
    def maximumWealth(self, accounts):
        max_wealth = 0

        for customer in accounts:
            total = 0
            for money in customer:
                total += money

            if total > max_wealth:
                max_wealth = total

        return max_wealth

4. Staircase Pattern

This program prints a right-aligned staircase pattern.

def staircase(n):
    for i in range(1, n+1):
        print(" " * (n - i) + "#" * i)

n = int(input())
staircase(n)

Example
Input: 6

Output
     #
    ##
   ###
  ####
 #####
######

6. Mini-Max Sum

This program finds the minimum and maximum sum of 4 elements out of 5.

def miniMaxSum(arr):
    total = sum(arr)
    min_val = min(arr)
    max_val = max(arr)

    print(total - max_val, total - min_val)


arr = list(map(int, input().split()))
miniMaxSum(arr)

Example
Input
1 2 3 4 5

Output
10 14


7. Birthday Cake Candles

This program counts the number of tallest candles.

def birthdayCakeCandles(candles):
    return candles.count(max(candles))


candles = list(map(int, input().split()))

8. Time Conversion

This program converts time from 12-hour format to 24-hour format.

def timeConversion(s):
    if s[-2:] == "AM" and s[:2] != "12":
        return s[:8]

    elif s[-2:] == "PM" and s[:2] != "12":
        return str(int(s[:2]) + 12) + s[2:8]

    elif s[:2] == "12" and s[-2:] == "AM":
        return "00" + s[2:8]

    else:
        return s[:8]


s = input()
print(timeConversion(s))


Example
Input
07:05:45PM

Output
19:05:45
